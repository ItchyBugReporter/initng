SET(WRAPPERS
	default
	runlevel
	virtual
	sh
	gentoo
)

FOREACH(WRAPPER ${WRAPPERS})
	CONFIGURE_FILE(
		"${CMAKE_CURRENT_SOURCE_DIR}/${WRAPPER}.in"
		"${CMAKE_CURRENT_BINARY_DIR}/${WRAPPER}"
		IMMEDIATE @ONLY)
ENDFOREACH(WRAPPER)

INSTALL_PROGRAMS(${CMAKE_INSTALL_PREFIX}${LIB_INSTALL_DIR}/initng/wrappers FILES ${WRAPPERS})
